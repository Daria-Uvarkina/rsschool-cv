                public class NextDate {
                    public static int getFebruaryDaysAmount(int year) {
                        if (((year % 4 != 0) || (year % 100 == 0)) && (year % 400 != 0)) {
                            return 28;
                        }
                
                        return 29;
                    }
                
                    public static int getDaysAmount(int month, int year) {
                        return switch (month) {
                            case 4, 6, 9, 11 -> 30;
                            case 2 -> getFebruaryDaysAmount(year);
                            default -> 31;
                        };
                    }
                
                    public static void main(String[] args) {
                        Scanner scanner = new Scanner(System.in);
                
                        System.out.println("Введите день:");
                        int day = scanner.nextInt();
                
                        System.out.println("Введите месяц:");
                        int month = scanner.nextInt();
                
                        System.out.println("Введите год:");
                        int year = scanner.nextInt();
                
                        int maximumDaysAmount = getDaysAmount(month, year);
                
                        if (month < 1 || month >= 13 || day < 1 || day > maximumDaysAmount) {
                            System.out.println("Некорректная дата");
                        } else {
                            day++;
                
                            if (day <= maximumDaysAmount) {
                                System.out.printf("Следующая дата - %02d.%02d.%d", day, month, year);
                            } else if (month < 12) {
                                day = 1;
                                month++;
                                System.out.printf("Следующая дата - %02d.%02d.%d", day, month, year);
                            } else {
                                day = 1;
                                month = 1;
                                year++;
                                System.out.printf("Следующая дата - %02d.%02d.%d", day, month, year);
                            }
                        }
